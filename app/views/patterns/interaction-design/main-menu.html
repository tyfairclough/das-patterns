Main Menu

We designed a top level navigation to facilitate users traversing through the product at any point.

Our product has non-linear task model and as such simple next steps don't always facilitate the user need. Our top menu gives users a broad sense where in the site they are and an easy, consistent way to move to other parts of the site.

The navigation has several features:

  • mobile first approach.
  • progressively enhances
  • priority menu disclosure

function progressiveMenuDisclosure() {
    $(".header-organisation").addClass("closed").removeClass("open");
    
  
        
        var viewportWidth = $(".header-organisation").outerWidth();
        var liSubTotalWidth = 0;
        var liTotalWidth = 0;
        var width = 0;
        var moreWidth = $(".more").outerWidth();
        var menuWidth = $(".menu").outerWidth();
        const mq = window.matchMedia( "(min-width: 641px)" );

        $(".header-inner li").each(function(){
            width = $(this).width();
            liSubTotalWidth = liSubTotalWidth + width;
        });    
        liTotalWidth = liSubTotalWidth - moreWidth - menuWidth;


        if ( mq.matches && (viewportWidth <= liTotalWidth) ) {
            $(".more").show();        
        } else {
            $(".more").hide();               
        }    
    
    $(".header-organisation .more, .header-organisation .menu").click(function(e){
        $(".header-organisation").addClass("open").removeClass("closed");
        $(".more").hide();
    });          
    }
    
    progressiveMenuDisclosure();
