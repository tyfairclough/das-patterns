           margin-bottom:100px;

}

// Share notification styles
.error-summary, .info-summary, .success-summary, .warning-summary {
    border-width: 4px;
    border-style: solid;

    margin-bottom: $gutter-half;
    padding: $gutter-half 10px;

    @include media(tablet) {
        border-width: 5px;
        border-style: solid;

        margin-top: $gutter;
        margin-bottom: $gutter;

        padding: $gutter-two-thirds $gutter-half $gutter-half;
    }

    @include ie-lte(6) {
        zoom: 1;
    }

    // Use the GOV.UK outline focus style
    &:focus {
        outline: 3px solid $focus-colour;
    }

    p {
        margin-bottom: 10px;
    }

    &-heading {
        margin-top: 0;
    }
    
    @for $index from 1 through 6 {
        h#{$index}{
        }

            margin: 0;
        }
    }
    
    &-list {
    padding-left: 0;
        li {
            @include media(tablet) {
            margin-bottom: 5px;
            }
        }
        a {
            font-weight: bold;
            text-decoration: underline;
        }
    }  
}
